PROTOCOLLO COMUNICAZIONE CLIENT SERVER
0 - carettere terminatore conversazione
1 - creazione nuovo dominio
2 - modifica durata prenotazione
3 - rinnovo prenotazione scaduta
4 - visualizza domini

--------------
status = "attivo"
	significa che il dominio è in funzione

status = "rinnovare"
	 significa che il dominio si trova in stallo per il rinnovo (bottone rosso)

status = "scaduto"
	significa che si ha scelto di non rinnovare il dominio
